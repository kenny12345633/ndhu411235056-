#include <vector>

using namespace std;

class Solution {
    vector<int> tmp; 

public:
    vector<int> sortArray(vector<int>& nums) {
        tmp.resize(nums.size());
        mergeSort(nums, 0, nums.size() - 1);
        return nums;
    }

private:
    void mergeSort(vector<int>& A, int p, int r) {
        if (p >= r) {
            return;
        }

        int q = (p + r) / 2; 

        mergeSort(A, p, q);

        mergeSort(A, q + 1, r);

        merge(A, p, q, r);
    }

    void merge(vector<int>& A, int p, int q, int r) {
        for (int i = p; i <= r; i++) {
            tmp[i] = A[i];
        }
        int i = p;     
        int j = q + 1; 
        int k = p;     

        while (i <= q && j <= r) {
            if (tmp[i] <= tmp[j]) { 
                A[k] = tmp[i];
                i++; 
            } else {              
                A[k] = tmp[j];
                j++; 
            }
            k++;
        }

        while (i <= q) {
            A[k] = tmp[i];
            i++;
            k++;
        }
    }
};
